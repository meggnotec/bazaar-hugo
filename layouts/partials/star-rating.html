{{- /* StarRating — 5-star SVG rating display with half-star support */ -}} {{- $rating := .rating | default 0.0 -}} {{- $reviewCount := .reviewCount -}} {{- $size := .size | default "sm" -}} {{- $starClass := "h-4 w-4" -}} {{- if eq $size "lg" -}} {{- $starClass = "h-6 w-6" -}} {{- else if eq $size "md" -}} {{- $starClass = "h-5 w-5" -}} {{- end -}} {{- $uid := printf "star-%d" (int (mul $rating 1000)) -}}
{{- range $i := seq 5 -}} {{- $starNum := float $i -}} {{- $ratingFloat := float $rating -}} {{- if le $starNum $ratingFloat -}} {{- /* Full star */ -}} {{- else if and (gt $ratingFloat (sub $starNum 1)) (lt $ratingFloat $starNum) -}} {{- /* Half star */ -}} {{- else -}} {{- /* Empty star */ -}} {{- end -}} {{- end }} {{- with $reviewCount }} ({{ . }}) {{- end }}
