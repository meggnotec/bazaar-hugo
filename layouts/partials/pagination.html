{{- /* Pagination — Page navigation with prev/next and page numbers */ -}} {{- /* Accepts either Hugo's built-in .Paginator or a dict with currentPage, totalPages, basePath */ -}} {{- $currentPage := .currentPage | default 1 -}} {{- $totalPages := .totalPages | default 1 -}} {{- $basePath := .basePath | default "/" -}} {{- if gt $totalPages 1 }}
{{- if gt $currentPage 1 }} {{- $prevPage := sub $currentPage 1 -}} ← Previous {{- end }} {{- range $page := seq $totalPages }} {{ $page }} {{- end }} {{- if lt $currentPage $totalPages }} {{- $nextPage := add $currentPage 1 -}} Next → {{- end }}
{{- end }}
